00051 void pal_trx_read_timestamp(uint32_t *timestamp)
00052 {
00053     /*
00054      * Everytime a transceiver interrupt is triggred, input capture register of
00055      * the AVR is latched. The 'sys_time' is concatenated to the ICR to
00056      * generate the time stamp of the received frame.
00057      * 'sys_time'   'ICR'
00058      *  ---------|--------- => 32 bit timestamp
00059      *   16 bits   16 bits
00060      */
00061     *timestamp = (uint32_t)sys_time << (uint32_t)16;
00062     *timestamp |= (uint32_t)TIME_STAMP_REGISTER;
00063 }

00110 void clock_init(void)
00111 {
00112     /*
00113      * Enable 32MHz internal oscillator
00114      * (and by thus disable the 2 MHz internal oscillator). */
00115     OSC.CTRL = OSC_RC32MEN_bm;
00116 
00117     /*
00118      * The ATxmega shall run from its internal 32MHz Oscillator.
00119      * Set the clock speed to 16MHz. Use internal 32MHz and DFLL.
00120      */
00121     while (0 == (OSC.STATUS & OSC_RC32MRDY_bm))
00122     {
00123         /* Hang until the internal 32MHz Oscillator is stable. */
00124         ;
00125     }
00126 
00127     /* Enable change of protected IO register */
00128     CCP = 0xD8;
00129     /* Use Prescaler A to divide 32MHz clock by 2 to 16MHz system clock. */
00130     CLK.PSCTRL = CLK_PSADIV0_bm;
00131 
00132     /* Enable change of protected IO register */
00133     CCP = 0xD8;
00134     /* Set internal 32MHz Oscillator as system clock. */
00135     CLK.CTRL = CLK_SCLKSEL0_bm;
00136 
00137     /* Enable DFLL for the internal oscillator. */
00138     OSC.DFLLCTRL = OSC_RC32MCREF_bm;
00139     DFLLRC32M.CTRL |= DFLL_ENABLE_bm;
00140 
00141 #ifdef SLEEPING_TIMER
00142     /* Turn on internal 32kHz. */
00143     OSC.CTRL |= OSC_RC32KEN_bm;
00144     do {
00145         /* Wait for the 32kHz oscillator to stabilize. */
00146     } while ( ( OSC.STATUS & OSC_RC32KRDY_bm ) == 0);
00147         /* Set internal 32kHz oscillator as clock source for RTC. */
00148     CLK.RTCCTRL = CLK_RTCSRC_RCOSC_gc | CLK_RTCEN_bm;
00149 #endif
00150 }

00173 /*
00174  * This function is called by timer_init() to perform the non-generic portion
00175  * of the initialization of the timer module.
00176  *
00177  * sys_clk = 16MHz -> Will be used as source for Event Channel 0 with Prescaler 16
00178  *
00179  * Timer usage
00180  * TCC0_CCA: Systime (software timer based on compare match)
00181  * TCC1_CCA: Input capture for time stamping
00182  */
00183 void timer_init_non_generic(void)
00184 {
00185     /* Select proper clock as timer clock source when radio is sleeping */
00186     TIMER_SRC_DURING_TRX_SLEEP();
00187 
00188     /*
00189      * Clear pending output compare matches of all.
00190      */
00191     TCC0_INTFLAGS = TC0_CCAIF_bm | TC0_CCBIF_bm | TC0_CCCIF_bm | TC0_CCDIF_bm;
00192     TCC1_INTFLAGS = TC1_CCAIF_bm | TC1_CCBIF_bm;
00193 
00194     /* Enable the timer overflow interrupt for TCC0 used for systime overflow. */
00195     TCC0_INTCTRLA =  TC_OVFINTLVL_HI_gc;
00196 
00197     /* Assign event channel 1 as input capture to TCC1_CCA */
00198     TCC1_CTRLB |= TC1_CCAEN_bm;
00199     TCC1_CTRLD = TC1_EVACT0_bm | TC1_EVSEL3_bm | TC1_EVSEL0_bm;
