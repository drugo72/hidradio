00169 static uint8_t eeprom_read_byte(uint8_t *addr)
00170 {
00171     uint16_t addrcpy;
00172 
00173     addrcpy = (uint16_t)addr;
00174 
00175     eeprom_wait_for_NVM();
00176 
00177     NVM.ADDR2 = 0;
00178     NVM.ADDR1 = (addrcpy >> 8) & 0xFF;
00179     NVM.ADDR0 = (addrcpy) & 0xFF;
00180     NVM.CMD =  NVM_CMD_READ_EEPROM_gc;
00181 
00182     /*
00183      * The CMDEX bit must be set within 4 clock cycles after setting the
00184      * protection byte in the CCP register.
00185      */
00186     NVM_EXEC();
00187     return NVM.DATA0;

00223 static void eeprom_write_byte(uint8_t addr, uint8_t value)
00224 {
00225     uint16_t addrcpy;
00226 
00227     addrcpy = (uint16_t)addr;
00228 
00229     /*  Flush buffer to make sure no unintetional data is written and load
00230      *  the "Page Load" command into the command register.
00231      */
00232     eeprom_flush_buffer();
00233     NVM.CMD = NVM_CMD_LOAD_EEPROM_BUFFER_gc;
00234 
00235     /* Set address to write to. */
00236     NVM.ADDR0 = addrcpy & 0xFF;
00237     NVM.ADDR1 = (addrcpy >> 8) & 0x1F;
00238     NVM.ADDR2 = 0x00;
00239 
00240     /* Load data to write, which triggers the loading of EEPROM page buffer. */
00241     NVM.DATA0 = value;
00242 
00243     /*  Issue EEPROM Atomic Write (Erase&Write) command. Load command, write
00244      *  the protection signature and execute command.
00245      */
00246     NVM.CMD = NVM_CMD_ERASE_WRITE_EEPROM_PAGE_gc;
00247     NVM_EXEC();
00248 }
